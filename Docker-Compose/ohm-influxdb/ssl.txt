# Выпуск самоподписанного сертификата в Linux
mkdir influxdb
cd influxdb
hostname=hv-us-01
ip=192.168.3.101
mkdir ssl
openssl req -x509 -newkey rsa:4096 -sha256 -days 3650 -nodes \
  -keyout ssl/ohm.key -out ssl/ohm.crt -subj "/CN=$hostname" \
  -addext "subjectAltName=DNS:$hostname,IP:$ip"
